<div style="width: 100%; height: 100%; overflow: hidden; position: relative;">
  <iframe id="myIframe" src="/web/album/index.html" style="width: 100%; height: 100%; border:none;"></iframe>
</div>

<script>
  function resizeIframe() {
    var iframe = document.getElementById("myIframe");
    if (iframe && iframe.contentWindow.document.body) {
      iframe.style.height = iframe.contentWindow.document.body.scrollHeight + "px"; // 콘텐츠 높이에 맞게
      iframe.style.width = "100%"; // 콘텐츠 너비에 맞게
    }
  }

  window.addEventListener("resize", resizeIframe); // 창 크기 변화 시 리사이즈

  document.getElementById("myIframe").onload = function() {
    setTimeout(resizeIframe, 100);
  };
</script>
